9)By yashmmith, contest: Arrays, problem: (Q) Count Quadraplets, Accepted, #, Copy
 #include <bits/stdc++.h>
using namespace std;
 
int main() {
    int no,x;
    cin >> no>>x;
    long long A[no];
    for (int i = 0; i < no; i++) {
        cin >> A[i];
    }
    int count=0;
    for(int i=0;i<no;i++){
        for(int j=i+1;j<no;j++){
            for(int k=j+1;k<no;k++){
                for(int l=k+1;l<no;l++){
                    if (A[i]-2*A[j]+3*A[k]-4*A[l]==x){
                        count++;
                    }
                }
            }
        }
    }
    cout<<count;
}
