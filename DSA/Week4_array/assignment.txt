9)By yashmmith, contest: Arrays, problem: (Q) Count Quadraplets, Accepted, #, Copy
 #include <bits/stdc++.h>
using namespace std;
 
int main() {
    int no,x;
    cin >> no>>x;
    long long A[no];
    for (int i = 0; i < no; i++) {
        cin >> A[i];
    }
    int count=0;
    for(int i=0;i<no;i++){
        for(int j=i+1;j<no;j++){
            for(int k=j+1;k<no;k++){
                for(int l=k+1;l<no;l++){
                    if (A[i]-2*A[j]+3*A[k]-4*A[l]==x){
                        count++;
                    }
                }
            }
        }
    }
    cout<<count;
}

11)#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;

        // Print odd numbers (increasing)
        for (int i = 1; i <= n; i += 2) {
            cout << i << " ";
        }

        // Print even numbers (decreasing)
        for (int i = n; i >= 2; i--) {
            if (i % 2 == 0) {
                cout << i << " ";
            }
        }

        cout << "\n";
    }

    return 0;
}





12)By yashmmith, contest: Arrays, problem: (K) Swap Alternate, Accepted, #, Copy
 #include <bits/stdc++.h>
using namespace std;
 
int main() {
    int no;
    cin >> no;
    for(int j=0;j<no;j++){
        int sn;
        cin>>sn;
        long long arr1[sn];
        for (int i = 0; i < sn; i++) {
            cin >> arr1[i];
        }
        int si=0,se=1;
        int size1=sizeof(arr1) / sizeof(arr1[0]);
        while(se<size1){
            int temp=arr1[si];
            arr1[si]=arr1[se];
            arr1[se]=temp;
            se+=2;
            si+=2;
        }
        for (int i = 0; i < sn; i++) {
            cout<< arr1[i]<<" ";
        }
        cout<<endl;
    }
}

        }

                }

        }

    }

17)#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while (t--) {
        int N;
        cin >> N;

        vector<int> A(N);
        for (int i = 0; i < N; i++) {
            cin >> A[i];
        }

        int M;
        cin >> M;

        unordered_map<int, int> freq;
        for (int i = 0; i < M; i++) {
            int x;
            cin >> x;
            freq[x]++;
        }

        bool printed = false;
        for (int i = 0; i < N; i++) {
            if (freq[A[i]] > 0) {
                cout << A[i] << " ";
                freq[A[i]]--;
                printed = true;
            }
        }

        if (!printed) {
            cout << "";
        }

        cout << "\n";
    }

    return 0;
}
