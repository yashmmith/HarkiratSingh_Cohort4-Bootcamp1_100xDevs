By yashmmith, contest: Functions, problem: (D) Check Prime, Accepted, #, Copy
 #include<bits/stdc++.h>
using namespace std;
#define int long long
 
bool pf(int no){
    int count=0;
    for (int i=1;i<=no;i++){
        if(no%i==0){
            count++;
        }
    }
    if(count==2){
        return 1;
    }
    else{
        return 0;
    }
}
 
signed main(){
    int no;
    cin>>no;
    int k=pf(no);
    if(k==1){
        cout<<"Prime";
    }
    else{
        cout<<"Not Prime";
    }
}

By yashmmith, contest: Functions, problem: (F) nCr, Accepted, #, Copy
 #include<bits/stdc++.h>
using namespace std;
#define int long long
 
signed main(){
    int n, r;
    cin >> n >> r;
 
    if (r > n) {
        cout << 0;
        return 0;
    }
 
    r = min(r, n - r); // optimization
    int res = 1;
 
    for (int i = 1; i <= r; i++) {
        res = res * (n - r + i) / i;
    }
 
    cout << res;
}

By yashmmith, contest: Functions, problem: (H) Count Zeros, Accepted, #, Copy
 #include <bits/stdc++.h>
using namespace std;
 
int main() {
    long long n;
    cin >> n;
    int count = 0;
    if (n == 0) { 
        cout << 1;
        return 0;
    }
    while (n > 0) {
        if (n % 10 == 0)
            count++;
        n /= 10;
    }
    cout << count;
}

 #include <bits/stdc++.h>
using namespace std;
 
int main() {
    long long a, b;
    cin >> a >> b;
 
    while (b != 0) {
        long long rem = a % b;
        a = b;
        b = rem;
    }
 
    cout << a;
}
