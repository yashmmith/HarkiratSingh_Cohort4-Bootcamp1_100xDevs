6.By yashmmith, contest: Loops, problem: (G) Count Numbers, Accepted, #, Copy
#include <bits/stdc++.h>
using namespace std;
 
int main() {
    int noc;
    cin>>noc;
    int positive=0,negative=0,even=0,odd=0;
    for (int i=0;i<noc;i++){
        int no;
        cin>>no;
        if (no>0){
            positive++;
        }
        else if(no<0){
            negative++;
        }
        if (no%2==0){
            even++;
        }
        else{
            odd++;
        }
    }
    cout<<positive<<endl;
    cout<<negative<<endl;
    cout<<even<<endl;
    cout<<odd<<endl;
}

9.By yashmmith, contest: Loops, problem: (J) $x^n$, Accepted, #, Copy
#include <bits/stdc++.h>
using namespace std;
 
int main() {
    long long n, m;
    cin >> n >> m;
 
    long long ans = 1;
    while (m > 0) {
        if (m & 1) ans *= n;
        n *= n;
        m >>= 1;
    }
 
    cout << ans;
    return 0;
}

11.By yashmmith, contest: Loops, problem: (M) Reverse Number and Store in a Variable, Accepted, #, Copy
#include <bits/stdc++.h>
using namespace std;
 
int main() {
    long long n;
    cin >> n;
    long long rev = 0;
    while (n > 0) {
        rev = rev * 10 + (n % 10);
        n /= 10;
    }
    cout << rev;
    return 0;
}
